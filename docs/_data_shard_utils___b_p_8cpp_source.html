90DEFINE_FUNCTION(UDataShardUtils_BP::execGetShardData)
91{
92 // Extract function parameters from the stack
93 P_GET_OBJECT(UObject, WorldContextObject);
94 P_GET_STRUCT(FDataShardKey, Key);
95
96 Stack.MostRecentPropertyAddress = nullptr;
97 Stack.StepCompiledIn<FProperty>(nullptr);
98 void* OutStruct = Stack.MostRecentPropertyAddress;
99 const FProperty* Property = Stack.MostRecentProperty;
100 P_FINISH;
101 GET_SHARD_MANAGER();
102
103 // Validate the key
104 if (!Manager->IsKeyValid(Key))
105 {
106 *(bool*)RESULT_PARAM = false;
107 Property->DestroyValue(OutStruct);
108 return;
109 }
110
111 // Validate data type compatibility
112 if (!Manager->IsDataShardSameType(Key, Property))
113 {
114 const FString ErrMsg = FString::Printf(
115 TEXT("Invalid data type. Shard: %s | Node: %s"),
116 *Manager->GetFieldClassFromShard(Key)->GetName(), *Property->GetClass()->GetName());
117 const FBlueprintExceptionInfo ExceptionInfo(EBlueprintExceptionType::AccessViolation,
118 FText::FromString(ErrMsg));
119 FBlueprintCoreDelegates::ThrowScriptException(P_THIS, Stack, ExceptionInfo);
120
121 *(bool*)RESULT_PARAM = false;
122 Property->DestroyValue(OutStruct);
123 return;
124 }
125
126 // Retrieve and copy the data
127 const void* Data = Manager->GetShardData<void>(Key);
128 const bool bResult = Data != nullptr;
129
130 Property->InitializeValue(OutStruct);
131 Property->CopyCompleteValue(OutStruct, Data);
132
133 *(bool*)RESULT_PARAM = bResult;
134}
